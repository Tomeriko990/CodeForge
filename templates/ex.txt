<script>
    const challenges = [
        { emoji: "🙋‍♀️✍️📖🫰🔛📦📦📦 🤔❓👉🧬🎭↔️🍸💊", answer: "הייתי כותבת ספר כותבת עושה עליו ארגזים", hint: "אודיה" },
        { emoji: "🗣️:🇮🇷🙅🏻‍♂️🔫➡️{🇺🇸+👨🏻‍🦰}🔜🔫🎼", answer: "איראן דונט שוט", hint: "פנינה רוזנבלום" },
        { emoji: "🏖️🌊🌞", answer: "חוף ים", hint: "מקום חופשה" },
        { emoji: "🧑‍💻💻", answer: "עבודה מרחוק", hint: "עבודה מהבית" }
    ];

    let currentChallengeIndex = 0;
    let attempts = 0;

    function normalize(str) {
        return str.replace(/\s+/g, ' ').trim(); // מנקה רווחים כפולים או מיותרים
    }

    function loadChallenge() {
        const currentChallenge = challenges[currentChallengeIndex];
        document.getElementById('emoji-display').textContent = currentChallenge.emoji;
        document.getElementById('user-input').value = '';
        document.getElementById('result-message').textContent = '';
        document.getElementById('try-again').textContent = '';
        document.getElementById('hint-button').style.display = 'none';
        hideHint();
    }

    function checkAnswer() {
        const userInput = normalize(document.getElementById('user-input').value);
        const correctAnswer = normalize(challenges[currentChallengeIndex].answer);
        
        if (userInput === correctAnswer) {
            alert("ברכות! הסרטן מתקרב בצעדי ענק 🦀");
            document.getElementById('result-message').textContent = "כל הכבוד! ימסכן!";
            document.getElementById('try-again').textContent = '';
            attempts = 0;
            currentChallengeIndex = (currentChallengeIndex + 1) % challenges.length;
            setTimeout(loadChallenge, 2000);
        } else {
            attempts++;
            document.getElementById('try-again').textContent = "נסה שנית!";
            if (attempts >= 3) {
                document.getElementById('hint-button').style.display = 'inline-block';
            }
        }
    }

    function giveHint() {
        const currentChallenge = challenges[currentChallengeIndex];
        document.getElementById('hint-text').textContent = "רמז: " + currentChallenge.hint;
        document.getElementById('hint-toast').classList.add("show");
    }

    function hideHint() {
        document.getElementById('hint-toast').classList.remove("show");
    }

    loadChallenge();
</script>
